Mysql(DB서버) 접속
cmd
mysql -uroot -p
1234

// 데이터베이스 목록
show databases;

//데이터베이스 상태
status

// 데이터베이스 전체내용 저장할 공간: 데이터베이스 생성
create database jspdb1;

// 데이터베이스 선택
use jspdb1;

// 주제별로 저장할 공간 => 테이블 생성
// 학생테이블 : 학생번호, 학생이름, 학생전화번호, 학생주소, 점수
//          : 기준값 설정 - 중복되지 않고, 비어있지 않게 설정
create table student(
	num int primary key,
 	name varchar(10)
);


// 테이블 목록 확인
show tables;

// 테이블 구조
desc student;
